242template<class value_type>
243void EllSparseBlockMat<value_type>::symv(SharedVectorTag, SerialTag, value_type alpha, const value_type* RESTRICT x, value_type beta, value_type* RESTRICT y) const
244{
245 //simplest implementation (all optimization must respect the order of operations)
246 for( int s=0; s<left_size; s++)
247 for( int i=0; i<num_rows; i++)
248 for( int k=0; k<n; k++)
249 for( int j=right_range[0]; j<right_range[1]; j++)
250 {
251 int I = ((s*num_rows + i)*n+k)*right_size+j;
252 // if y[I] isnan then even beta = 0 does not make it 0
253 y[I] = beta == 0 ? (value_type)0 : y[I]*beta;
254 for( int d=0; d<blocks_per_line; d++)
255 {
256 value_type temp = 0;
257 for( int q=0; q<n; q++) //multiplication-loop
258 temp = DG_FMA( data[ (data_idx[i*blocks_per_line+d]*n + k)*n+q],
259 x[((s*num_cols + cols_idx[i*blocks_per_line+d])*n+q)*right_size+j],
260 temp);
261 y[I] = DG_FMA( alpha,temp, y[I]);
262 }
263 }
