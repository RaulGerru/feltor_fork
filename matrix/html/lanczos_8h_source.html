294 m_TH.values(i,2) = betaip; // +1 diagonal
295

334 template<class UnaryOp>
335 value_type compute_universal_error( const HDiaMatrix& TH, unsigned iter,
336 unsigned q, UnaryOp f, HVec& yH)
337 {
338 unsigned new_iter = iter + 1 + q;
339 set_iter( iter+1);
340 HDiaMatrix THtilde( new_iter, new_iter, 3*new_iter-2, 3);
341 THtilde.diagonal_offsets[0] = -1;
342 THtilde.diagonal_offsets[1] = 0;
343 THtilde.diagonal_offsets[2] = 1;
344 for( unsigned u=0; u<iter+1; u++)
345 {
346 THtilde.values(u,0) = TH.values(u,0);
347 THtilde.values(u,1) = TH.values(u,1);
348 THtilde.values(u,2) = TH.values(u,2);
349 }
350 for( unsigned u=1; u<=q; u++)
351 {
352 THtilde.values( iter+u, 0) = u==1 ? TH.values(iter,2) :
353 TH.values( iter+1-u, 1);
354 THtilde.values( iter+u, 1) = TH.values( iter-u, 1);
355 THtilde.values( iter+u, 2) = TH.values( iter-u, 0);
356 }
357 yH = f( TH);
358 HVec yHtilde = f( THtilde);
359 for( unsigned u=0; u<yH.size(); u++)
360 yHtilde[u] -= yH[u];
361 value_type norm = dg::fast_l2norm( yHtilde);
362 return norm;
363 }
