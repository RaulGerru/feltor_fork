55template<class value_type>
56void compute_Tinv_y( const cusp::dia_matrix<int, value_type,
57 cusp::host_memory>& T,
58 thrust::host_vector<value_type>& x,
59 const thrust::host_vector<value_type>& y, value_type a = 1.,
60 value_type d = 0.)
61{
62 unsigned size = y.size();
63 x.resize(size);
64 thrust::host_vector<value_type> ci(size), di(size);
65 ci[0] = a*T.values(0,2)/( a*T.values(0,1) + d);
66 di[0] = y[0]/( a*T.values(0,1) + d);
67 for( unsigned i=1; i<size; i++)
68 {

72 }
73 x[size-1] = di[size-1];
74 for( int i=size-2; i>=0; i--)
75 x[i] = di[i] - ci[i]*x[i+1];

167 template<class ContainerType0, class ContainerType1, class ContainerType2>
168 void operator()(const ContainerType0& a, const ContainerType1& b,
169 const ContainerType2& c, CooMatrix& Tinv)
170 {
171 unsigned ss = m_size;
172 Tinv.resize(ss, ss, ss* ss);
173 if( ss == 1)
174 {
175 Tinv.row_indices[0] = 0;
176 Tinv.column_indices[0] = 0;
177 Tinv.values[0] = 1./b[0];
178 return;
179 }
180 //fill alphas
181 m_alphas[0]=1.0;
182 m_alphas[1]=a[0];
183 for( unsigned i = 2; i<ss+1; i++)
184 {
185 m_alphas[i] = a[i-1]*m_alphas[i-1] - c[i-1]*b[i-2]*m_alphas[i-2];
186 if (m_alphas[i] ==0 && i<ss) {
187 throw dg::Error( dg::Message(_ping_) << "# Failure in alpha["<<i<<"] !");
188 }
189 }
190 if (m_alphas[ss] ==0)
191 throw dg::Error( dg::Message(_ping_) << "# No Inverse of tridiagonal matrix exists !");
192
193 //fill betas
194 m_betas[ss]=1.0;
195 m_betas[ss-1]=a[ss-1];
196 m_betas[0] = m_alphas[ss];
197 for( int i = ss-2; i>0; i--)
198 {
199 m_betas[i] = a[i]*m_betas[i+1] - c[i+1]*b[i]*m_betas[i+2];
200 if (m_betas[i] ==0)
201 {
202 throw dg::Error( dg::Message(_ping_) << "# Failure in beta["<<i<<"] !");
203 }
204 }
205 //Diagonal entries
206 Tinv.row_indices[0*ss+0] = 0;
207 Tinv.column_indices[0*ss+0] = 0;
208 Tinv.row_indices[(ss-1)*ss+(ss-1)] = (ss-1);
209 Tinv.column_indices[(ss-1)*ss+(ss-1)] = (ss-1);
210 Tinv.values[0*ss+0] = 1.0/(a[0]-c[1]*b[0]*m_betas[2]/m_betas[1]);
211 Tinv.values[(ss-1)*ss+(ss-1)] = 1.0/(a[ss-1] -
212 c[ss-1]*b[ss-2]*m_alphas[ss-2]/m_alphas[ss-1]);
213 for( unsigned i=1; i<ss-1; i++)
214 {
215 Tinv.row_indices[i*ss+i] = i;
216 Tinv.column_indices[i*ss+i] = i;
217 Tinv.values[i*ss+i] =
218 1.0/(a[i]-c[i]*b[i-1]*m_alphas[i-1]/m_alphas[i]
219 -c[i+1]*b[i]*m_betas[i+2]/m_betas[i+1]);
220 }
221 //Off-diagonal entries
222 for( unsigned i=0; i<ss; i++)
223 {
224 for( unsigned j=0; j<ss; j++)
225 {
226 Tinv.row_indices[i*ss+j] = j;
227 Tinv.column_indices[i*ss+j] = i;
228 if (i<j) {
229 Tinv.values[i*ss+j] =
230 sign(j-i)*std::accumulate(std::next(b.begin(),i),
231 std::next(b.begin(),j), 1.,
232 std::multiplies<value_type>())*
233 m_alphas[i]/m_alphas[j]*Tinv.values[j*ss+j];
234 }
235 else if (i>j)
236 {
237 Tinv.values[i*ss+j] =
238 sign(i-j)*std::accumulate(std::next(c.begin(),j+1),
239 std::next(c.begin(),i+1), 1.,
240 std::multiplies<value_type>())*
241 m_betas[i+1]/m_betas[j+1]*Tinv.values[j*ss+j];
242 }
243 }
244 }
245 }

508 template<class ContainerType0, class ContainerType1, class ContainerType2>
509 void operator()(const ContainerType0& a, const ContainerType1& b,
510 const ContainerType2& c, CooMatrix& Tinv)
511 {
512 Tinv.resize( m_size, m_size, m_size*m_size);
513 unsigned is=0;
514 for( unsigned i = 0; i<m_size+1; i++)
515 {
516 is = m_size - i;
517 if (i==0)
518 {
519 m_theta[0] = 1.;
520 m_phi[is] = 1.;
521 }
522 else if (i==1)
523 {
524 m_theta[1] = a[0];
525 m_phi[is] = a[is];
526 }
527 else
528 {
529 m_theta[i] = a[i-1] * m_theta[i-1] - b[i-2] * c[i-1] * m_theta[i-2];
530 m_phi[is] = a[is] * m_phi[is+1] - b[is] * c[is+1] * m_phi[is+2];
531 }
532 }
533
534 //Compute inverse tridiagonal matrix elements
535 for( unsigned i=0; i<m_size; i++) //row index
536 {
537 for( unsigned j=0; j<m_size; j++) //column index

545 std::multiplies<value_type>())*sign(i+j) *
